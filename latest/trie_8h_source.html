244  template<typename ValueType>
245  [[nodiscard]] Trie<ValueType> Trie<ValueType>::get_subtrie(const Address &address, bool strict) const {
246  if (address.empty()) {
247  return *this; // calls copy constructor, returns Trie with shared data
248  }
249  if (has_value()) {
250  throw TrieKeyError(address);
251  }
252  auto iter = map_->find(address.first());
253  if (iter == map_->end()) {
254  if (strict) {
255  throw TrieKeyError(address);
256  } else {
257  return Trie<ValueType>{}; // return new empty choice trie
258  }
259  } else {
260  const Address &rest = address.rest();
261  if (rest.empty()) {
262  return iter->second;
263  } else {
264  try {
265  return iter->second.get_subtrie(rest, strict);
266  } catch (const TrieKeyError &) {
267  throw TrieKeyError(address);
268  }
269  }
270  }

273  template<typename ValueType>
274  void Trie<ValueType>::set_subtrie(const Address &address, Trie<ValueType> trie, bool overwrite) {
275  if (address.empty()) {
276  if (!overwrite && !empty()) {
277  throw TrieOverwriteError(address);
278  }
279  // copy the shared pointers
280  value_ = trie.value_;
281  map_ = trie.map_;
282  } else {
283  value_->reset(); // delete choice value, if there is one
284  auto iter = map_->find(address.first());
285  Trie<ValueType> *rest_subtrie_ptr = nullptr;
286  if (iter == map_->end()) {
287  // there is no subtrie at this key, add it
288  Trie<ValueType> rest_subtrie{};
289  map_->insert({address.first(), std::move(rest_subtrie)});
290  rest_subtrie_ptr = &((*map_)[address.first()]);
291  } else {
292  // there is a subtrie at this key
293  rest_subtrie_ptr = &iter->second; // copy assignment
294  }
295  try {
296  rest_subtrie_ptr->set_subtrie(address.rest(), std::move(trie), overwrite);
297  } catch (const TrieOverwriteError &) {
298  throw TrieOverwriteError(address);
299  }
300  };

308  inline void pretty_print_value(std::ostream &out, const std::any &value) {
309  if (const auto *tensor = std::any_cast<Tensor>(&value)) {
310  if (tensor->ndimension() == 0) {
311  // a scalar
312  if (tensor->dtype() == torch::ScalarType::Float) {
313  out << tensor->item<float>();
314  } else if (tensor->dtype() == torch::ScalarType::Double) {
315  out << tensor->item<double>();
316  } else if (tensor->dtype() == torch::ScalarType::Bool) {
317  out << (tensor->item<bool>() ? "true" : "false");
318  } else {
319  std::stringstream ss;
320  ss << "printing choices not implemented for Tensor dtype: ";
321  ss << tensor->dtype();
322  throw std::logic_error(ss.str()); // TODO implement support for this
323  }
324  } else {
325  // not a scalar
326  out << "Tensor"; // TODO give a better representation
327  }
328  } else if (const auto *str = std::any_cast<std::string>(&value)) {
329  out << *str;
330  } else if (const auto *str = std::any_cast<const char *>(&value)) {
331  out << *str;
332  } else if (const auto *i = std::any_cast<int>(&value)) {
333  out << *i;
334  } else {
335  throw std::logic_error("Could not pretty-print choice value"); // TODO give a better error msg
336  }
337  }

368  template<typename ValueType>
369  std::ostream &Trie<ValueType>::pretty_print(std::ostream &out, int pre, const std::vector<int> &vert_bars,
370  bool extra_space) const {
371  if (has_value()) {
372  pretty_print_value(out, get_value());
373  return out;
374  }
375  auto indent_vert = get_indent_vert(pre, vert_bars);
376  auto indent_vert_last = get_indent_vert_last(pre, vert_bars);
377  auto indent = get_indent(pre, vert_bars);
378  auto indent_last = get_indent_last(pre, vert_bars);
379  size_t cur = 0;
380  size_t n = map_->size();
381  for (const auto &key_and_subtrie: *map_) {
382  bool is_last = (cur == n - 1);
383  std::string key = key_and_subtrie.first;
384  if (extra_space) {
385  out << indent_vert;
386  }
387  if (is_last) {
388  out << indent_last;
389  } else {
390  out << indent;
391  }
392  if (key_and_subtrie.second.has_value()) {
393  out << key << " : ";
394  pretty_print_value(out, key_and_subtrie.second.get_value());
395  out << std::endl;
396  } else {
397  out << key << std::endl;
398  std::vector<int> next_vert_bars = vert_bars;
399  if (!is_last) {
400  next_vert_bars.push_back(pre);
401  }
402  key_and_subtrie.second.pretty_print(out, pre + 4, next_vert_bars);
403  }
404  cur += 1;
405  }
406  return out;
407  }
