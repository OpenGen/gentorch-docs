278 template<typename ValueType>
279  [[nodiscard]] Trie<ValueType> Trie<ValueType>::get_subtrie(const Address &address, bool strict) const {
280  if (address.empty()) {
281  return *this; // calls copy constructor, returns Trie with shared data
282  }
283  if (has_value()) {
284  throw TrieKeyError(address);
285  }
286  auto iter = map_->find(address.first());
287  if (iter == map_->end()) {
288  if (strict) {
289  throw TrieKeyError(address);
290  } else {
291  return Trie<ValueType>{}; // return new empty choice trie
292  }
293  } else {
294  const Address &rest = address.rest();
295  if (rest.empty()) {
296  return iter->second;
297  } else {
298  try {
299  return iter->second.get_subtrie(rest, strict);
300  } catch (const TrieKeyError &) {
301  throw TrieKeyError(address);
302  }
303  }
304  }
305  }

307  template<typename ValueType>
308  void Trie<ValueType>::set_subtrie(const Address &address, Trie<ValueType> trie, bool overwrite) {
309  if (address.empty()) {
310  if (!overwrite && !empty()) {
311  throw TrieOverwriteError(address);
312  }
313  // copy the shared pointers
314  value_ = trie.value_;
315  map_ = trie.map_;
316  } else {
317  value_->reset(); // delete choice value, if there is one
318  auto iter = map_->find(address.first());
319  Trie<ValueType> *rest_subtrie_ptr = nullptr;
320  if (iter == map_->end()) {
321  // there is no subtrie at this key, add it
322  Trie<ValueType> rest_subtrie{};
323  map_->insert({address.first(), std::move(rest_subtrie)});
324  rest_subtrie_ptr = &((*map_)[address.first()]);
325  } else {
326  // there is a subtrie at this key
327  rest_subtrie_ptr = &iter->second; // copy assignment
328  }
329  try {
330  rest_subtrie_ptr->set_subtrie(address.rest(), std::move(trie), overwrite);
331  } catch (const TrieOverwriteError &) {
332  throw TrieOverwriteError(address);
333  }
334  };

342  inline void pretty_print_value(std::ostream &out, const std::any &value) {
343  if (const auto *tensor = std::any_cast<Tensor>(&value)) {
344  if (tensor->ndimension() == 0) {
345  // a scalar
346  if (tensor->dtype() == torch::ScalarType::Float) {
347  out << tensor->item<float>();
348  } else if (tensor->dtype() == torch::ScalarType::Double) {
349  out << tensor->item<double>();
350  } else if (tensor->dtype() == torch::ScalarType::Bool) {
351  out << (tensor->item<bool>() ? "true" : "false");
352  } else {
353  std::stringstream ss;
354  ss << "printing choices not implemented for Tensor dtype: ";
355  ss << tensor->dtype();
356  throw std::logic_error(ss.str()); // TODO implement support for this
357  }
358  } else {
359  // not a scalar
360  out << "Tensor"; // TODO give a better representation
361  }
362  } else if (const auto *str = std::any_cast<std::string>(&value)) {
363  out << *str;
364  } else if (const auto *str = std::any_cast<const char *>(&value)) {
365  out << *str;
366  } else if (const auto *i = std::any_cast<int>(&value)) {
367  out << *i;
368  } else {
369  throw std::logic_error("Could not pretty-print choice value"); // TODO give a better error msg
370  }
371  }

402  template<typename ValueType>
403  std::ostream &Trie<ValueType>::pretty_print(std::ostream &out, int pre, const std::vector<int> &vert_bars,
404  bool extra_space) const {
405  if (has_value()) {
406  pretty_print_value(out, get_value());
407  return out;
408  }
409  auto indent_vert = get_indent_vert(pre, vert_bars);
410  auto indent_vert_last = get_indent_vert_last(pre, vert_bars);
411  auto indent = get_indent(pre, vert_bars);
412  auto indent_last = get_indent_last(pre, vert_bars);
413  size_t cur = 0;
414  size_t n = map_->size();
415  for (const auto &key_and_subtrie: *map_) {
416  bool is_last = (cur == n - 1);
417  std::string key = key_and_subtrie.first;
418  if (extra_space) {
419  out << indent_vert;
420  }
421  if (is_last) {
422  out << indent_last;
423  } else {
424  out << indent;
425  }
426  if (key_and_subtrie.second.has_value()) {
427  out << key << " : ";
428  pretty_print_value(out, key_and_subtrie.second.get_value());
429  out << std::endl;
430  } else {
431  out << key << std::endl;
432  std::vector<int> next_vert_bars = vert_bars;
433  if (!is_last) {
434  next_vert_bars.push_back(pre);
435  }
436  key_and_subtrie.second.pretty_print(out, pre + 4, next_vert_bars);
437  }
438  cur += 1;
439  }
440  return out;
441  }
