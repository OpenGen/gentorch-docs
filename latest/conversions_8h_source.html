77  template<typename T, typename U>
78  pair<size_t, pair<T, U>> roll(const vector<Tensor> &unrolled, size_t start, const pair<T, U> &value) {
79  auto[after_first, rolled_first] = roll(unrolled, start, value.first);
80  auto[after_second, rolled_second] = roll(unrolled, after_first, value.second);
81  return {after_second, {rolled_first, rolled_second}};

95  template<typename T>
96  pair<size_t, vector<T>> roll(const vector<Tensor> &unrolled, size_t start, const vector<T> &value) {
97  vector<T> rolled;
98  for (const auto element: value) {
99  auto[new_start, rolled_element] = roll(unrolled, start, element);
100  start = new_start;
101  rolled.emplace_back(rolled_element);
102  }
103  return {start, rolled};
104  }
