69 template<typename args_type, typename return_value_type, typename subtrace_type>
70 struct MyGradNode : public Node {
71  explicit MyGradNode(subtrace_type& subtrace, const GradientHelper& helper, edge_list &&next_edges)
72  : Node(std::move(next_edges)), subtrace_{subtrace},
73  helper_{helper} {};
74 
75  ~MyGradNode() override = default;
76 
77  variable_list apply(variable_list &&output_grad) override {
78  const return_value_type& return_value = subtrace_.get_return_value();
79 
80  // read off the return value gradient from the output_grad
81  auto [num_read, return_value_grad] = roll(output_grad, 0, return_value);
82  // check that we read off all but the last element, which is the dummy element
83  assert(num_read == output_grad.size() - 1);
84 
85  args_type args_grad = subtrace_.parameter_gradient(
86  *helper_.accumulator_ptr_, return_value_grad, *helper_.scaler_ptr_);
87  std::vector<Tensor> input_grad = unroll(args_grad);
88  input_grad.emplace_back(torch::tensor(0.0));
89  return input_grad;
90  }
91 
92 private:
93  subtrace_type& subtrace_;
94  const GradientHelper& helper_;
95 };

97 template<typename args_type, typename return_type, typename subtrace_type>
98 struct MyNode : public Node {
99  explicit MyNode(subtrace_type& subtrace, const GradientHelper& helper)
100  : subtrace_{subtrace}, helper_{helper} {}
101 
102  ~MyNode() override = default;
103 
104  variable_list apply(variable_list &&inputs) override {
105  const return_type& value = subtrace_.get_return_value();
106  vector<Tensor> output = unroll(value);
107 
108  // value that we use to ensure the corresponding MyGradNode is visited in the backward pass
109  Tensor dummy = torch::tensor(0.0); // TODO optimize me, maybe make static?
110  output.emplace_back(dummy);
111 
112  return torch::autograd::wrap_outputs(inputs, std::move(output), [&](edge_list &&next_edges) {
113  return std::make_shared<MyGradNode<args_type, return_type, subtrace_type>>(subtrace_, helper_, std::move(next_edges));
114  });
115  }
116 
117 private:
118  subtrace_type& subtrace_;
119  const GradientHelper& helper_;
120 };

144 template<typename T>
145 T detach_clone_and_track(const T &args) {
146  assert(!c10::InferenceMode::is_enabled());
147  vector<Tensor> args_unrolled = unroll(args);
148  vector<Tensor> args_unrolled_copy;
149  for (auto &t: args_unrolled) {
150  args_unrolled_copy.emplace_back(t.detach().clone().set_requires_grad(true));
151  }
152  return roll(args_unrolled_copy, args); // copy elision
153 }

260  args_type parameter_gradient(GradientAccumulator& accumulator, return_type retval_grad, double scaler) {
261 
262  // the computation graph is already set up, there should be no Tensors created anyways
263  c10::InferenceMode guard {true};
264 
265  if (!prepared_for_gradients_) {
266  throw std::logic_error("not ready for gradients");
267  }
268 
269  // add the arguments to the inputs
270  vector<Tensor> inputs = unroll(get_args());
271  size_t num_args = inputs.size();
272 
273  // add the local parameters to the inputs
274  vector<Tensor> local_parameter_tensors = parameters_.local_parameters();
275  size_t num_local_parameters = local_parameter_tensors.size();
276  inputs.insert(inputs.end(), local_parameter_tensors.begin(), local_parameter_tensors.end());
277 
278  // add the dummy input
279  assert(dummy_input_.requires_grad());
280  inputs.emplace_back(dummy_input_);
281 
282  // add the return value to the outputs
283  const return_type& retval = get_return_value();
284  vector<Tensor> outputs = unroll(retval);
285  vector<Tensor> output_grads = unroll(retval_grad);
286 
287  // add the dummy output
288  assert(dummy_output_.requires_grad());
289  outputs.emplace_back(dummy_output_);
290  output_grads.emplace_back(torch::tensor(1.0));
291 
292  // pass information back to the calls to parameter_gradient() that happen within the backward pass will have
293  // NOTE: not thread safe, but this is okay, traces aren't intended to be thread safe
294  helper_->scaler_ptr_ = &scaler;
295  helper_->accumulator_ptr_ = &accumulator;
296 
297  // do the backward pass. this recursively invokes parameter_gradient() for each subtrace
298  // NOTE: the returned input grads do not include parameters for callee generative functions
299  // TODO check why we are setting allow_unused to true
300  vector<Tensor> input_grads = torch::autograd::grad(outputs, inputs, output_grads, {}, false, true);
301 
302  // read off argument gradients
303  vector<Tensor> args_grad_unrolled;
304  size_t i;
305  for (i = 0; i < num_args; ++i) {
306  args_grad_unrolled.emplace_back(input_grads[i]);
307  }
308  args_type args_grad = roll(args_grad_unrolled, get_args());
309 
310  // read off local parameter gradients and increment the accumulators
311  assert(i == num_args);
312  for (auto it = accumulator.begin(parameters_); it != accumulator.end(parameters_); ++it) {
313  assert((*it).sizes().equals(input_grads[i].sizes()));
314  (*it).add_(input_grads[i++], scaler);
315  }
316 
317  // the last element should be the dummy
318  assert(i == input_grads.size() - 1);
319 
320  return std::move(args_grad);
321  }
322 
323  args_type parameter_gradient(GradientAccumulator& accumulator, double scaler) {
324  return parameter_gradient(accumulator, zero_gradient(get_return_value()), scaler);
325  }

327  template <typename args_type, typename return_type, typename subtrace_type>
328  return_type make_tracked_return_value(subtrace_type& subtrace, const args_type& tracked_args, const return_type& value) {
329  auto node = MyNode<args_type, return_type, subtrace_type>(subtrace, *helper_);
330  // NOTE: gen_fn_with_args.get_args() returns args that are tracked as part of the autograd graph
331  std::vector<Tensor> inputs = unroll(tracked_args);
332  assert(dummy_input_.requires_grad());
333  inputs.emplace_back(dummy_input_);
334  auto outputs = node(std::move(inputs));
335  auto [num_read, tracked_value] = roll(outputs, 0, value);
336  assert(num_read == outputs.size() - 1);
337  Tensor dummy = outputs[num_read];
338  assert(dummy.requires_grad());
339  dummy_output_ += dummy;
340  assert(dummy_output_.requires_grad());
341  return tracked_value;
342  }

345  pair<const args_type&, std::unique_ptr<const args_type>> args_;
346  std::unique_ptr<Trie<std::unique_ptr<Trace>>> subtraces_;
347  ChoiceTrie choices_;
348  double score_;
349  optional<return_type> maybe_value_;
350  bool prepared_for_gradients_;
351  bool assert_retval_grad_;
352  std::unique_ptr<GradientHelper> helper_;
353  parameters_type& parameters_;
354  Tensor dummy_input_;
355  Tensor dummy_output_;
356 };
