32 template<typename GenFnType>
33 class PrimitiveTrace : public Trace {
34 public:
35  typedef typename GenFnType::return_type return_type;
36  typedef typename GenFnType::args_type args_type;
37  typedef typename GenFnType::dist_type dist_type;
38 
39  PrimitiveTrace(return_type &&value, const dist_type &dist) :
40  value_{std::make_unique<return_type>(value)},
41  value_alternate_{std::make_unique<return_type>(value)},
42  dist_{std::make_unique<dist_type>(dist)},
43  dist_alternate_{std::make_unique<dist_type>(dist)},
44  score_{dist_->log_density(*value_)} {}
45 
46  [[nodiscard]] const return_type& return_value() const {
47  return *value_;
48  }
49 
50  [[nodiscard]] args_type parameter_gradient(
51  GradientAccumulator& accumulator, const return_type&, double scaler) {
52  return parameter_gradient(accumulator, scaler);
53  }
54 
55  [[nodiscard]] args_type parameter_gradient(GradientAccumulator& accumulator, double scaler) {
56  auto grads = dist_->log_density_gradient(*value_);
57  return GenFnType::extract_argument_gradient(grads);
58  }
59 
60  [[nodiscard]] double score() const override {
61  return score_;
62  }
63 
64  [[nodiscard]] ChoiceTrie choices() const override {
65  ChoiceTrie trie{};
66  trie.set_value(*value_);
67  return trie; // copy elision
68  }

70  template<typename RNG>
71  double update(RNG& rng, const gentl::change::UnknownChange<GenFnType>& change,
72  const ChoiceTrie& constraints, const UpdateOptions& options) {
73  const GenFnType& new_gen_fn = change.new_value();
74  double log_weight;
75  const return_type& prev_value = *value_.get();
76  double prev_score = score_;
77  if (options.save()) {
78  std::swap(value_, value_alternate_);
79  std::swap(dist_, dist_alternate_);
80  std::swap(score_, score_alternate_);
81  can_be_reverted_ = true;
82  }
83  if (constraints.has_value()) {
84  backward_constraints_.set_value(prev_value);
85  *value_ = std::any_cast<return_type>(constraints.get_value());
86  *dist_ = new_gen_fn.dist_;
87  score_ = dist_->log_density(*value_);
88  log_weight = score_ - prev_score;
89  } else if (constraints.empty()) {
90  log_weight = 0.0;
91  backward_constraints_.clear();
92  } else {
93  throw std::domain_error("expected primitive or empty choice dict");
94  }
95  has_backward_constraints_ = true;
96  return log_weight;
97  }
98 
99  [[nodiscard]] const ChoiceTrie& backward_constraints() const {
100  if (!has_backward_constraints_)
101  throw std::logic_error("there are no backward constraints; call update");
102  return backward_constraints_;
103  }
104 
105  void revert() override {
106  if (!can_be_reverted_)
107  throw std::logic_error("cannot be reverted; call update with save=true");
108  std::swap(value_, value_alternate_);
109  std::swap(dist_, dist_alternate_);
110  std::swap(score_, score_alternate_);
111  can_be_reverted_ = false;
112  has_backward_constraints_ = false;
113  }
114 
115  std::unique_ptr<PrimitiveTrace<GenFnType>> fork() {
116  // calls copy constructors for return_type and dist_type
117  auto trace = std::unique_ptr<PrimitiveTrace<GenFnType>>(
118  new PrimitiveTrace<GenFnType>(return_type{*this->value_}, *this->dist_));
119  trace->score_ = this->score_;
120  return std::move(trace);
121  }
122 
123  std::unique_ptr<Trace> fork_type_erased() override {
124  return fork();
125  }
126 
127 private:
128  std::unique_ptr<return_type> value_;
129  std::unique_ptr<return_type> value_alternate_;
130  std::unique_ptr<dist_type> dist_;
131  std::unique_ptr<dist_type> dist_alternate_;
132  ChoiceTrie backward_constraints_;
133  double score_;
134  double score_alternate_;
135  bool can_be_reverted_{false};
136  bool has_backward_constraints_{false};
137 };

175  template<class RNGType>
176  std::pair<std::unique_ptr<trace_type>, double>
177  generate(RNGType &rng, const EmptyModule& parameters, const ChoiceTrie& constraints,
178  const GenerateOptions&) const {
179  return_type value;
180  double log_weight;
181  if (constraints.has_value()) {
182  value = std::any_cast<return_type>(constraints.get_value());
183  log_weight = dist_.log_density(value);
184  } else if (constraints.empty()) {
185  value = dist_.sample(rng);
186  log_weight = 0.0;
187  } else {
188  throw std::domain_error("expected primitive or empty choice dict");
189  }
190  return {std::make_unique<trace_type>(std::move(value), dist_), log_weight};
191  }
